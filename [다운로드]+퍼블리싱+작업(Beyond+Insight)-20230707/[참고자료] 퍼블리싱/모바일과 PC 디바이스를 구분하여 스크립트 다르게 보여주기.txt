■ 모바일/PC 디바이스 구분하여 스크립트 다르게 보여주기

아래의 방법은 디바이스 크기가 아니라 디바이스 기종에 따라서 모바일과 모바일이 아닌 경우를 구분하는 스크립트입니다. 그래서 PC에서 화면을 작게한다해도 모바일 스크립트가 적용되지 않습니다.(새로고침할 경우 적용됨)

기종이 Android|webOS|iPhone|iPad|iPod|BlackBerry인 경우를 모바일 디바이스로 인식합니다.

$(function(){
    /* 모바일 기기 체크 */
    var isMobile = /Android|webOS|iPhone|iPad|iPod|BlackBerry/i.test(navigator.userAgent) ? true : false;
    if(!isMobile) {  
        // 모바일인 경우 스크립트
    } else {
        // 모바일이 아닌 경우 스크립트
    }
})

■ 모바일/PC 디바이스 너비를 기준으로 스크립트 다르게 보여주기

아래의 방법은 디바이스 종류가 아니라 디바이스 너비를 기준을로 모바일과 모바일이 아닌 경우를 구분하는 스크립트입니다. 그래서 PC에서 화면의 너비를 모바일 크기로 작게 하는 경우 모바일 스크립트가 적용됩니다.(반응형 웹 제작을 위해 사용하기 좋습니다.)

디바이스의 크기는 제작자의 결정에 따라 수치를 조정합니다.

$(window).resize(function(){ 
    /* 다바이스 크기가 480 이상 일때 */
    if (window.innerWidth > 480) {
         // 모바일이 아닌 경우 스크립트
    } else {
         // 모바일인 경우 스크립트
    }
}).resize();
